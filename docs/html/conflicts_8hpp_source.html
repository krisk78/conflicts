137 template <typename T>
138 bool Conflicts<T>::deep_search(const T& object1, const T& object2, const T* prev) const noexcept
139 {
140 bool result = m_conflicts.exists(object1, object2) || m_conflicts.exists(object2, object1);
141 if (!result && m_cascading)
142 {
143 // Deep search : if 2 objects are in conflict then a third one that is in conflict with one of them is in conflict with the other
144 auto requirements = m_conflicts.requirements(object1);
145 auto req = requirements.begin();
146 while (!result && req != requirements.end())
147 {
148 if (prev == nullptr || !(*req == *prev || *req == object2))
149 result = deep_search(*req, object2, &object1);
150 ++req;
151 }
152 if (!result)
153 {
154 auto dependents = m_conflicts.dependents(object1);
155 auto dep = dependents.begin();
156 while (!result && dep != dependents.end())
157 {
158 if (prev == nullptr || !(*dep == *prev || *dep == object2))
159 result = deep_search(*dep, object2, &object1);
160 ++dep;
161 }
162 }
163 }
164 return result;
165 }

202 {
203 std::vector<T> result{};
204 auto confs = m_conflicts.requirements(object);
205 for (auto con : confs)
206 {
207 if (prev == nullptr || !(con == *prev))
208 {
209 result.push_back(con);
210 // perform a recursive search
211 auto res = all_conflicts(con, &object);
212 for (auto itr : res)
213 result.push_back(itr);
214 }
215 }
216 confs = m_conflicts.dependents(object);
217 for (auto con : confs)
218 {
219 if (prev == nullptr || !(con == *prev))

221 result.push_back(con);
222 // perform a recursive search
223 auto res = all_conflicts(con, &object);
224 for (auto itr : res)
225 result.push_back(itr);
226 }
227 }
228 return result;
